-- Enable necessary extensions
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
CREATE EXTENSION IF NOT EXISTS "pgvector";

-- Define ENUM types for status fields
CREATE TYPE client_status_enum AS ENUM ('Active', 'Inactive');
CREATE TYPE project_status_enum AS ENUM ('SOW_REVIEW', 'MATCHING', 'PROPOSAL', 'ACTIVE', 'COMPLETED', 'CANCELLED');
CREATE TYPE sow_document_status_enum AS ENUM ('PROCESSING', 'REVIEW_PENDING', 'FAILED', 'COMPLETED');
CREATE TYPE project_brief_status_enum AS ENUM ('PENDING_REVIEW', 'APPROVED');
CREATE TYPE proposal_status_enum AS ENUM ('SUBMITTED', 'SHORTLISTED', 'REJECTED', 'AWARDED');
CREATE TYPE invoice_status_enum AS ENUM ('DRAFT', 'SENT', 'PAID', 'OVERDUE');
CREATE TYPE payout_status_enum AS ENUM ('PENDING', 'PAID', 'FAILED');
CREATE TYPE audit_log_action_type_enum AS ENUM ('CREATE', 'UPDATE', 'DELETE', 'OVERRIDE');

-- Table: Role
CREATE TABLE "Role" (
    "roleId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "name" VARCHAR(50) NOT NULL,
    CONSTRAINT "UC_Role_Name" UNIQUE ("name")
);

-- Table: Client
CREATE TABLE "Client" (
    "clientId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "companyName" VARCHAR(255) NOT NULL,
    "status" client_status_enum NOT NULL DEFAULT 'Active',
    "isDeleted" BOOLEAN NOT NULL DEFAULT false,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Table: Vendor
CREATE TABLE "Vendor" (
    "vendorId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "companyName" VARCHAR(255) NOT NULL,
    "contactInfo" JSONB,
    "paymentDetails" JSONB,
    "isDeleted" BOOLEAN NOT NULL DEFAULT false,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Table: Skill
CREATE TABLE "Skill" (
    "skillId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "name" VARCHAR(100) NOT NULL,
    "vectorEmbedding" VECTOR, -- Assuming pgvector extension is used. Specify size, e.g., VECTOR(1536)
    CONSTRAINT "UC_Skill_Name" UNIQUE ("name")
);

-- Table: User
CREATE TABLE "User" (
    "userId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "email" VARCHAR(255) NOT NULL,
    "passwordHash" VARCHAR(255) NOT NULL,
    "firstName" VARCHAR(100) NOT NULL,
    "lastName" VARCHAR(100) NOT NULL,
    "roleId" UUID NOT NULL,
    "clientId" UUID,
    "vendorId" UUID,
    "isActive" BOOLEAN NOT NULL DEFAULT true,
    "isDeleted" BOOLEAN NOT NULL DEFAULT false,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "UC_User_Email" UNIQUE ("email")
);

-- Table: VendorSkill (Join Table)
CREATE TABLE "VendorSkill" (
    "vendorId" UUID NOT NULL,
    "skillId" UUID NOT NULL,
    PRIMARY KEY ("vendorId", "skillId")
);

-- Table: Project
CREATE TABLE "Project" (
    "projectId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "clientId" UUID NOT NULL,
    "name" VARCHAR(255) NOT NULL,
    "status" project_status_enum NOT NULL DEFAULT 'SOW_REVIEW',
    "marginOverride" DECIMAL(5, 2),
    "totalInvoicedAmount" DECIMAL(14, 2) NOT NULL DEFAULT 0.00,
    "totalPaidOutAmount" DECIMAL(14, 2) NOT NULL DEFAULT 0.00,
    "calculatedProfit" DECIMAL(14, 2) NOT NULL DEFAULT 0.00,
    "isDeleted" BOOLEAN NOT NULL DEFAULT false,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Table: SowDocument
CREATE TABLE "SowDocument" (
    "sowDocumentId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "originalFilename" VARCHAR(255) NOT NULL,
    "storagePath" VARCHAR(1024) NOT NULL,
    "status" sow_document_status_enum NOT NULL DEFAULT 'PROCESSING',
    "processingErrorDetails" TEXT,
    "uploadedById" UUID NOT NULL,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "UC_SowDocument_ProjectId" UNIQUE ("projectId")
);

-- Table: ProjectBrief
CREATE TABLE "ProjectBrief" (
    "projectBriefId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "scopeSummary" TEXT,
    "status" project_brief_status_enum NOT NULL DEFAULT 'PENDING_REVIEW',
    "reviewedById" UUID,
    "reviewedAt" TIMESTAMPTZ,
    CONSTRAINT "UC_ProjectBrief_ProjectId" UNIQUE ("projectId")
);

-- Table: ProjectBriefSkill (Join Table)
CREATE TABLE "ProjectBriefSkill" (
    "projectBriefId" UUID NOT NULL,
    "skillId" UUID NOT NULL,
    PRIMARY KEY ("projectBriefId", "skillId")
);

-- Table: Proposal
CREATE TABLE "Proposal" (
    "proposalId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "vendorId" UUID NOT NULL,
    "cost" DECIMAL(12, 2) NOT NULL,
    "timelineDescription" TEXT NOT NULL,
    "status" proposal_status_enum NOT NULL DEFAULT 'SUBMITTED',
    "submittedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "UC_Proposal_Project_Vendor" UNIQUE ("projectId", "vendorId")
);

-- Table: ProposalQuestion
CREATE TABLE "ProposalQuestion" (
    "proposalQuestionId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "submittedByVendorId" UUID NOT NULL,
    "questionText" TEXT NOT NULL,
    "answerText" TEXT,
    "answeredById" UUID,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "answeredAt" TIMESTAMPTZ
);

-- Table: Invoice (Partitioned)
CREATE TABLE "Invoice" (
    "invoiceId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "amount" DECIMAL(12, 2) NOT NULL,
    "status" invoice_status_enum NOT NULL DEFAULT 'DRAFT',
    "issuedDate" TIMESTAMPTZ NOT NULL,
    "dueDate" TIMESTAMPTZ NOT NULL
) PARTITION BY RANGE ("issuedDate");
-- Note: Partitions must be created manually, e.g.:
-- CREATE TABLE "Invoice_y2024" PARTITION OF "Invoice" FOR VALUES FROM ('2024-01-01') TO ('2025-01-01');

-- Table: Payout
CREATE TABLE "Payout" (
    "payoutId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "vendorId" UUID NOT NULL,
    "amount" DECIMAL(12, 2) NOT NULL,
    "status" payout_status_enum NOT NULL DEFAULT 'PENDING',
    "paidDate" TIMESTAMPTZ
);

-- Table: ProjectPayoutRule
CREATE TABLE "ProjectPayoutRule" (
    "projectPayoutRuleId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "projectId" UUID NOT NULL,
    "description" VARCHAR(255) NOT NULL,
    "payoutPercentage" DECIMAL(5, 2)
);

-- Table: AuditLog (Partitioned)
CREATE TABLE "AuditLog" (
    "auditLogId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "actorUserId" UUID NOT NULL,
    "actionType" audit_log_action_type_enum NOT NULL,
    "entityName" VARCHAR(100) NOT NULL,
    "entityId" VARCHAR(36) NOT NULL,
    "changes" JSONB NOT NULL,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
) PARTITION BY RANGE ("createdAt");
-- Note: Partitions must be created manually, e.g.:
-- CREATE TABLE "AuditLog_2024_01" PARTITION OF "AuditLog" FOR VALUES FROM ('2024-01-01') TO ('2024-02-01');

-- Table: SystemSetting
CREATE TABLE "SystemSetting" (
    "settingKey" VARCHAR(100) PRIMARY KEY,
    "settingValue" JSONB NOT NULL,
    "description" TEXT
);

-- Table: Notification
CREATE TABLE "Notification" (
    "notificationId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "recipientUserId" UUID NOT NULL,
    "eventType" VARCHAR(100) NOT NULL,
    "content" TEXT NOT NULL,
    "relatedEntityName" VARCHAR(100),
    "relatedEntityId" UUID,
    "isRead" BOOLEAN NOT NULL DEFAULT false,
    "readAt" TIMESTAMPTZ,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- Table: WebhookEndpoint
CREATE TABLE "WebhookEndpoint" (
    "webhookEndpointId" UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    "url" VARCHAR(2048) NOT NULL,
    "description" TEXT,
    "secretToken" VARCHAR(255),
    "isActive" BOOLEAN NOT NULL DEFAULT true,
    "createdAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT "UC_WebhookEndpoint_Url" UNIQUE ("url")
);

-- Table: DashboardMetrics
CREATE TABLE "DashboardMetrics" (
    "metricKey" VARCHAR(50) PRIMARY KEY DEFAULT 'main',
    "projectsByStatusCount" JSONB,
    "pendingActionsCount" JSONB,
    "financialMetrics" JSONB,
    "upcomingMilestones" JSONB,
    "lastUpdatedAt" TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX "IX_AuditLog_Actor" ON "AuditLog" ("actorUserId");
CREATE INDEX "IX_AuditLog_CreatedAt_EntityName" ON "AuditLog" ("createdAt" DESC, "entityName");
CREATE INDEX "IX_Notification_Recipient_Unread" ON "Notification" ("recipientUserId", "isRead");

CREATE INDEX "IX_WebhookEndpoint_IsActive" ON "WebhookEndpoint" ("isActive");
